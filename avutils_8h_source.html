324 template<typename T, typename L, typename C>
325 T guessValue(const T& value, const L * list, C endListComparator)
326 {
327  if (!list)
328  return value;
329 
330  // move values to array
331  std::deque<T> values;
332  for (const L * ptr = list; !endListComparator(*ptr); ++ptr)
333  {
334  T v = *ptr;
335  values.push_back(v);
336  }
337 
338  // sort list
339  std::sort(values.begin(), values.end());
340 
341  // Search more appropriate range
342  int begin = 0;
343  int end = values.size() - 1;
344  while ((end - begin) > 1)
345  {
346  int mid = begin + (end - begin) / 2;
347 
348  if (value <= values[mid])
349  {
350  end = mid;
351  }
352  else
353  {
354  begin = mid;
355  }
356  }
357 
358  // distance from VALUE to BEGIN more short or VALUE less then BEGIN
359  if (value <= values[begin] || (value - values[begin]) < (values[end] - value))
360  {
361  return values[begin];
362  }
363 
364  return values[end];
365 }
